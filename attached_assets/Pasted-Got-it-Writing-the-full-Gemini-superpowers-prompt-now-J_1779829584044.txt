PART 1 — STREAMING RESPONSES
What This Does
Currently Jarvis waits for the complete Gemini response before speaking. With streaming, Gemini sends tokens as they generate. The first sentence reaches TTS in under 1 second. Conversations feel dramatically more alive.
Architecture Change
Current flow:
User speaks → Gemini API → wait 2-3s for full response → TTS speaks full text
New flow:
User speaks → Gemini streaming API → 
first sentence detected → TTS starts speaking immediately →
second sentence detected → queued → 
third sentence → queued →
all while Gemini is still generating
Implementation in src/lib/gemini.ts
Add a new method streamGenerate to the GeminiAPI class:
typescriptasync *streamGenerate(
  messages: GeminiMessage[],
  systemPrompt: string,
  tools?: object[]
): AsyncGenerator<string, void, unknown> {
  // Use streamGenerateContent endpoint
  const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:streamGenerateContent?alt=sse&key=${this.primaryKey}`

  const payload = {
    contents: messages,
    systemInstruction: {
      parts: [{ text: systemPrompt }]
    },
    ...(tools && tools.length > 0 ? { tools } : {}),
    generationConfig: {
      temperature: 0.7,
      topK: 40,
      topP: 0.95,
      maxOutputTokens: 8192,
      thinkingConfig: {
        thinkingBudget: 0 // disable thinking for streaming — adds latency
      }
    }
  }

  let response: Response
  
  try {
    response = await fetch(url, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload),
    })
  } catch {
    // Network error — fall back to primary key failure handler
    throw new Error('Network error during streaming request')
  }

  if (!response.ok) {
    // On 429 or 403 — retry with fallback key
    if (response.status === 429 || response.status === 403) {
      yield* this._streamWithKey(messages, systemPrompt, tools, this.fallbackKey)
      return
    }
    throw new Error(`Gemini streaming error: ${response.status}`)
  }

  yield* this._streamWithKey(messages, systemPrompt, tools, this.primaryKey)
}

private async *_streamWithKey(
  messages: GeminiMessage[],
  systemPrompt: string,
  tools: object[] | undefined,
  apiKey: string
): AsyncGenerator<string, void, unknown> {
  const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:streamGenerateContent?alt=sse&key=${apiKey}`

  const payload = {
    contents: messages,
    systemInstruction: { parts: [{ text: systemPrompt }] },
    ...(tools && tools.length > 0 ? { tools } : {}),
    generationConfig: {
      temperature: 0.7,
      topK: 40,
      topP: 0.95,
      maxOutputTokens: 8192,
      thinkingConfig: { thinkingBudget: 0 }
    }
  }

  const response = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload),
  })

  if (!response.ok || !response.body) {
    throw new Error(`Streaming failed: ${response.status}`)
  }

  const reader = response.body.getReader()
  const decoder = new TextDecoder()

  try {
    while (true) {
      const { done, value } = await reader.read()
      if (done) break

      const chunk = decoder.decode(value, { stream: true })
      const lines = chunk.split('\n')

      for (const line of lines) {
        if (!line.startsWith('data: ')) continue
        const data = line.slice(6).trim()
        if (data === '[DONE]') return

        try {
          const parsed = JSON.parse(data)
          const text = parsed
            ?.candidates?.[0]
            ?.content
            ?.parts
            ?.find((p: { text?: string }) => p.text)
            ?.text

          if (text) yield text
        } catch {
          // Malformed SSE chunk — skip
        }
      }
    }
  } finally {
    reader.releaseLock()
  }
}
Sentence Chunker
Add a utility class to src/lib/gemini.ts that accumulates streaming tokens and emits complete sentences:
typescriptexport class SentenceChunker {
  private buffer = ''
  private readonly SENTENCE_ENDINGS = /[.!?]\s+|[.!?]$/

  // Returns complete sentences found in the buffer, keeps remainder
  push(token: string): string[] {
    this.buffer += token
    const sentences: string[] = []

    let match: RegExpExecArray | null
    let lastIndex = 0

    // Find all sentence boundaries
    const regex = /[.!?](\s+|$)/g
    while ((match = regex.exec(this.buffer)) !== null) {
      const sentence = this.buffer.slice(lastIndex, match.index + 1).trim()
      if (sentence.length > 0) sentences.push(sentence)
      lastIndex = match.index + match[0].length
    }

    // Keep remainder after last sentence boundary
    this.buffer = this.buffer.slice(lastIndex)
    return sentences
  }

  // Call at end of stream to flush any remaining text
  flush(): string[] {
    const remainder = this.buffer.trim()
    this.buffer = ''
    return remainder.length > 0 ? [remainder] : []
  }

  reset(): void {
    this.buffer = ''
  }
}

PART 4 — CONTEXT CACHING
What This Does
The system prompt (personality + memories + timeline context) is rebuilt and sent with every single message — consuming hundreds of tokens each time. Context caching tells Gemini to store it once and reuse it. Reduces token consumption by ~60% and speeds up responses.
Implementation in src/lib/gemini.ts
Add a CacheManager class:
typescriptinterface CachedContext {
  cacheId: string
  createdAt: number
  expiresAt: number
  systemPrompt: string
}

class CacheManager {
  private cache: CachedContext | null = null
  private readonly TTL_MS = 60 * 60 * 1000 // 1 hour

  isValid(systemPrompt: string): boolean {
    if (!this.cache) return false
    if (Date.now() > this.cache.expiresAt) return false
    if (this.cache.systemPrompt !== systemPrompt) return false
    return true
  }

  getCacheId(): string | null {
    return this.cache?.cacheId ?? null
  }

  set(cacheId: string, systemPrompt: string): void {
    this.cache = {
      cacheId,
      createdAt: Date.now(),
      expiresAt: Date.now() + this.TTL_MS,
      systemPrompt,
    }
  }

  invalidate(): void {
    this.cache = null
  }
}

const cacheManager = new CacheManager()
Add a createContextCache method to GeminiAPI:
typescriptasync createContextCache(
  systemPrompt: string,
  apiKey: string
): Promise<string | null> {
  try {
    const url = `https://generativelanguage.googleapis.com/v1beta/cachedContents?key=${apiKey}`

    const payload = {
      model: 'models/gemini-2.5-flash',
      systemInstruction: {
        parts: [{ text: systemPrompt }]
      },
      ttl: '3600s',
    }

    const response = await fetch(url, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload),
    })

    if (!response.ok) return null

    const data = await response.json()
    return data.name ?? null  // cache ID like "cachedContents/abc123"
  } catch {
    return null
  }
}

NON-NEGOTIABLE REQUIREMENTS

Streaming only activates for conversation intent — tool calls and search still use existing non-streaming paths
SentenceChunker must be exported from gemini.ts and imported in jarvis.ts
The streaming generator must release the reader lock in a finally block — memory leak if not
thinkingBudget: 0 must be set on the streaming endpoint — thinking mode + streaming causes latency issues
Thinking budget 1024 only for queries matching requiresDeepThinking — not all queries
CODE_EXECUTION_TOOL and functionDeclarations tools are mutually exclusive — never send both in the same request
CacheManager is module-level singleton — not a class that gets instantiated per request
Cache TTL is 1 hour — matches Gemini's minimum cached content lifetime
If cache creation fails — silently fall back to sending full system prompt — never throw
All existing methods (generate, generateWithTools, generateWithSearch) must continue working exactly as before — streaming is additive
Zero TypeScript errors
Zero any types anywhere in new code
