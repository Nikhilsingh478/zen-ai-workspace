PART 4 — WEBSOCKET TEARDOWN FIX (PERMANENT)
File: src/lib/horizon.ts
Problem: WebSocket connection closed before established — appears on every load. Subscription setup and teardown is happening too fast on rapid mount/unmount cycles.
Fix: Find setupRealtime(). Wrap with mounted guard and delay:
typescriptlet _realtimeMounted = true
let _realtimeTimer: ReturnType<typeof setTimeout> | null = null
let _realtimeChannel: ReturnType<typeof supabase.channel> | null = null

function setupRealtime() {
  _realtimeMounted = true
  _realtimeTimer = setTimeout(() => {
    if (!_realtimeMounted) return
    _realtimeChannel = supabase
      .channel('horizon_tasks_realtime')
      .on(
        'postgres_changes',
        { event: '*', schema: 'public', table: 'horizon_tasks' },
        (payload) => {
          if (!_realtimeMounted) return
          // existing handler logic unchanged
        }
      )
      .subscribe()
  }, 150)
}

function teardownRealtime() {
  _realtimeMounted = false
  if (_realtimeTimer) {
    clearTimeout(_realtimeTimer)
    _realtimeTimer = null
  }
  if (_realtimeChannel) {
    supabase.removeChannel(_realtimeChannel)
    _realtimeChannel = null
  }
}

PART 5 — AUDIO STATE MACHINE: STUCK IN RESPONDING + NO IDLE RETURN
File: src/lib/jarvis.ts
Problem 1: After speaking finishes, state gets stuck in speaking/responding forever. Browser TTS onend event is unreliable — Chrome has a known bug where speechSynthesis pauses after ~15 seconds and onend never fires.
Fix: Replace the entire TTSQueue.playNext() method:
typescriptprivate playNext(): void {
  if (this.queue.length === 0) {
    this.isPlaying = false
    transitionAudioState('idle')
    return
  }

  const sentence = this.queue.shift()!
  const utterance = new SpeechSynthesisUtterance(sentence)
  utterance.rate = 1.0
  utterance.pitch = 1.0

  // Estimated duration + buffer for timeout fallback
  const estimatedMs = Math.max((sentence.length / 12) * 1000, 3000) + 4000
  let timeoutId: ReturnType<typeof setTimeout> | null = setTimeout(() => {
    window.speechSynthesis.cancel()
    this.queue = []
    this.isPlaying = false
    transitionAudioState('idle')
  }, estimatedMs)

  const cleanup = () => {
    if (timeoutId) {
      clearTimeout(timeoutId)
      timeoutId = null
    }
  }

  utterance.onend = () => {
    cleanup()
    this.playNext()
  }

  utterance.onerror = () => {
    cleanup()
    this.isPlaying = false
    transitionAudioState('idle')
  }

  window.speechSynthesis.speak(utterance)
}
Problem 2: Chrome speechSynthesis pauses mid-speech after ~15s.
Fix: Add keepalive in TTSQueue:
typescriptprivate startKeepAlive(): void {
  this.keepAliveInterval = setInterval(() => {
    if (window.speechSynthesis.speaking && !window.speechSynthesis.paused) {
      window.speechSynthesis.pause()
      window.speechSynthesis.resume()
    }
  }, 10000)
}

private stopKeepAlive(): void {
  if (this.keepAliveInterval) {
    clearInterval(this.keepAliveInterval)
    this.keepAliveInterval = null
  }
}
Call startKeepAlive() when first sentence starts playing, stopKeepAlive() when queue empties or is interrupted.

PART 6 — INTERRUPT WORDS NOT WORKING
File: src/lib/jarvis.ts
Problem: "stop", "wait" etc don't interrupt Jarvis while speaking because no SpeechRecognition instance is listening during the speaking state.
Fix: Add a dedicated interrupt listener that activates whenever state transitions to speaking:
typescriptlet _interruptRecognition: SpeechRecognition | null = null

function startInterruptListener(): void {
  if (!('SpeechRecognition' in window || 'webkitSpeechRecognition' in window)) return
  try {
    const SR = (window as any).SpeechRecognition || (window as any).webkitSpeechRecognition
    _interruptRecognition = new SR()
    _interruptRecognition!.continuous = false
    _interruptRecognition!.interimResults = false
    _interruptRecognition!.lang = 'en-US'
    _interruptRecognition!.maxAlternatives = 1

    _interruptRecognition!.onresult = (event: SpeechRecognitionEvent) => {
      const transcript = event.results[0][0].transcript.toLowerCase().trim()
      const INTERRUPT_WORDS = ['stop', 'wait', 'hold on', 'shut up', 'pause', 'enough', 'quiet']
      if (INTERRUPT_WORDS.some(word => transcript.includes(word))) {
        ttsQueue.interrupt()
        stopInterruptListener()
        transitionAudioState('interrupted')
        setTimeout(() => {
          if (_state.audioState === 'interrupted') {
            transitionAudioState('idle')
          }
        }, 800)
      }
    }

    _interruptRecognition!.onerror = () => {
      // silent fail — interrupt listener is best-effort
    }

    _interruptRecognition!.onend = () => {
      // restart if still speaking
      if (_state.audioState === 'speaking') {
        try { _interruptRecognition?.start() } catch (e) {}
      }
    }

    _interruptRecognition!.start()
  } catch (e) {
    console.warn('[JARVIS] Interrupt listener failed to start:', e)
  }
}

function stopInterruptListener(): void {
  if (_interruptRecognition) {
    try { _interruptRecognition.stop() } catch (e) {}
    _interruptRecognition = null
  }
}
In transitionAudioState():

When transitioning TO speaking → call startInterruptListener()
When transitioning AWAY from speaking → call stopInterruptListener()


PART 7 — MARKDOWN ASTERISKS BEING READ ALOUD
File: src/lib/jarvis.ts
Problem: Gemini responses contain markdown (**bold**, *italic*, ##headings, bullet points) which get passed raw to TTS and read aloud as "asterisk asterisk bold asterisk asterisk".
Fix: Add cleanTextForSpeech() function:
typescriptfunction cleanTextForSpeech(text: string): string {
  return text
    .replace(/\*\*(.*?)\*\*/g, '$1')
    .replace(/\*(.*?)\*/g, '$1')
    .replace(/#{1,6}\s+/g, '')
    .replace(/`{3}[\s\S]*?`{3}/g, 'code block omitted')
    .replace(/`([^`]+)`/g, '$1')
    .replace(/\[([^\]]+)\]\([^)]+\)/g, '$1')
    .replace(/^\s*[-*+]\s+/gm, '')
    .replace(/^\s*\d+\.\s+/gm, '')
    .replace(/\n{2,}/g, '. ')
    .replace(/\n/g, ' ')
    .replace(/\s{2,}/g, ' ')
    .trim()
}
Call this on EVERY text string before passing to ttsQueue.enqueue(). No exceptions — morning briefing, search results, regular replies, task confirmations, all of it.
