PART B — loadSessionMessages IN src/lib/jarvis.ts
Check if loadSessionMessages is already exported. If it is, skip this section. If it is not, add it now:
typescript/**
 * Loads all messages for a specific past session from Supabase.
 * Returns them formatted as DisplayMessage objects for rendering.
 */
export async function loadSessionMessages(
  sessionId: string
): Promise<DisplayMessage[]> {
  try {
    const { data, error } = await supabase
      .from('jarvis_messages')
      .select('*')
      .eq('session_id', sessionId)
      .order('created_at', { ascending: true })

    if (error) {
      console.error('[JARVIS] loadSessionMessages error:', error)
      return []
    }

    if (!data || data.length === 0) return []

    return data.map(row => ({
      id: row.id as string,
      role: row.role as 'user' | 'assistant',
      content: row.content as string,
      type: (row.message_type as string) ?? 'conversation',
      timestamp: new Date(row.created_at as string),
    }))
  } catch (err) {
    console.error('[JARVIS] loadSessionMessages unexpected error:', err)
    return []
  }
}

typescriptconst handleHistoryCardClick = useCallback(async (session: {
  id: string
  started_at: string
  message_count: number
  session_summary: string | null
}) => {
  if (historyViewLoading) return

  setHistoryViewLoading(true)

  try {
    const messages = await loadSessionMessages(session.id)

    setHistoryViewSession({
      id: session.id,
      startedAt: new Date(session.started_at),
      messageCount: session.message_count,
      summary: session.session_summary,
    })

    setHistoryViewMessages(messages)

    // Use the same cinematic animation sequence from Part 1

  } catch (err) {
    console.error('[JARVIS] Failed to load session messages:', err)
  } finally {
    setHistoryViewLoading(false)
  }
}, [historyViewLoading, triggerExtendedWindowSequence])

const handleHistoryViewClose = useCallback(async () => {
