typescriptfunction getBestMaleVoice(): SpeechSynthesisVoice | null {
  const voices = window.speechSynthesis.getVoices()
  
  // Priority order for male voices
  const malePriority = [
    'Google UK English Male',
    'Microsoft David Desktop',
    'Microsoft Mark Desktop',
    'Alex',                      // macOS male
    'Daniel',                    // macOS UK male
    'Google US English',         // Usually male
  ]
  
  for (const name of malePriority) {
    const match = voices.find(v => v.name.includes(name))
    if (match) return match
  }
  
  // Fallback — find any voice with 'male' in the name
  const maleByName = voices.find(v => v.name.toLowerCase().includes('male'))
  if (maleByName) return maleByName
  
  // Last resort — first English voice
  return voices.find(v => v.lang.startsWith('en')) ?? null
}

FIX 3 — OpenWakeWord: load() Before start() Race Condition
Problem: [WakeWord] Failed to initialize: Error: Call load() before start() — the implementation calls engine.start() before engine.load() resolves. This is a race condition.
Fix: In src/lib/jarvis.ts, in startWakeWordDetection(), ensure strict sequential async execution:
typescriptasync function startWakeWordDetection(): Promise<void> {
  if (isMobile()) return
  if (_wakeWordActive || _wakeWordEngine) return

  try {
    _wakeWordEngine = new WakeWordEngine({
      baseAssetUrl: '/openwakeword/models',
      keywords: ['hey_jarvis'],
      detectionThreshold: 0.35, // LOWERED from 0.5 — more sensitive, picks up normal voice
      cooldownMs: 2500,
    })

    // CRITICAL: Must fully await load() before calling start()
    // Do NOT call start() in a .then() chain — await sequentially
    self.postMessage?.({ type: 'status', msg: 'Loading wake word model...' })
    await _wakeWordEngine.load()
    
    // Only wire up events AFTER load completes
    _wakeWordEngine.on('detect', ({ keyword, score }: { keyword: string, score: number }) => {
      console.log(`[WakeWord] Detected: ${keyword} (score: ${score.toFixed(2)})`)
      if (_state.audioState === 'idle' && _state.enabled) {
        transitionAudioState('listening')
        startCommandListening()
      }
    })

    _wakeWordEngine.on('error', (err: Error) => {
      console.warn('[WakeWord] Runtime error:', err.message)
    })

    // Only call start() after load() AND event wiring is complete
    await _wakeWordEngine.start()
    _wakeWordActive = true

    console.log('[JARVIS] Wake word active — say "Hey Jarvis" in normal voice')
  } catch (err) {
    console.warn('[WakeWord] Init failed — tap-to-talk still works:', (err as Error).message)
    _wakeWordEngine = null
    _wakeWordActive = false
  }
}

let _briefingDeliveredThisSession = false

export function shouldDeliverMorningBriefing(): boolean {
  if (_briefingDeliveredThisSession) return false

  const today = new Date().toISOString().split('T')[0]
