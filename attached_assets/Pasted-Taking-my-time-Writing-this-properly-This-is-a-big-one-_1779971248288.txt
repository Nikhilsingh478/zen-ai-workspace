```typescript
// Module-level proactive check interval
let _proactiveCheckInterval: ReturnType<typeof setInterval> | null = null

export function startProactiveChecks(): void {
  if (_proactiveCheckInterval) return

  // Run immediately on start, then every 15 minutes
  runProactiveCheck()
  _proactiveCheckInterval = setInterval(runProactiveCheck, 15 * 60 * 1000)
}

export function stopProactiveChecks(): void {
  if (_proactiveCheckInterval) {
    clearInterval(_proactiveCheckInterval)
    _proactiveCheckInterval = null
  }
}

async function runProactiveCheck(): Promise<void> {
  try {
    const now = new Date()
    const todayStr = now.toLocaleDateString('en-CA')
    const currentHour = now.getHours()
    const currentMinute = now.getMinutes()

    const { data: tasks } = await supabase
      .from('horizon_tasks')
      .select('*')
      .eq('task_date', todayStr)
      .eq('completed', false)
      .order('task_time', { ascending: true })

    if (!tasks || tasks.length === 0) return

    for (const task of tasks) {
      const [taskHour, taskMin] = (task.task_time as string)
        .split(':').map(Number)
      const taskMins = taskHour * 60 + taskMin
      const nowMins = currentHour * 60 + currentMinute
      const diff = taskMins - nowMins

      // 15 minute warning
      if (diff >= 13 && diff <= 17) {
        addProactiveMessage(
          `Heads up — "${task.title}" starts in 15 minutes.`
        )
        return // one proactive message at a time
      }

      // High priority overdue
      if (
        task.priority === 'high' &&
        nowMins > taskMins + 30
      ) {
        addProactiveMessage(
          `"${task.title}" was scheduled for ${task.task_time} and hasn't been marked complete.`
        )
        return
      }
    }

    // Evening check — 7pm nothing done
    if (currentHour === 19 && currentMinute < 15) {
      const { count } = await supabase
        .from('horizon_tasks')
        .select('*', { count: 'exact', head: true })
        .eq('task_date', todayStr)
        .eq('completed', true)

      if ((count ?? 0) === 0) {
        addProactiveMessage(
          `It's 7pm and you haven't completed any tasks today. ${tasks.length} still pending.`
        )
      }
    }
  } catch {
    // Silent — proactive checks must never crash the app
  }
}

function addProactiveMessage(text: string): void {
  // Add as a special proactive message type to JARVIS state
  // Does NOT go through TTS — visual only, shown in chat feed
  patch({
    messages: [
      ..._state.messages,
      {
        id: `proactive_${Date.now()}`,
        role: 'assistant' as const,
        content: text,
        type: 'proactive_alert',
        timestamp: new Date(),
      }
    ]
  })
}
```

## FEATURE 4 — HABIT TRACKING AWARENESS

### What It Does
Jarvis analyzes your last 7 days of Horizon task completion by domain and surfaces patterns. Injected into the system prompt so Jarvis references it naturally in conversation.

### Step 1 — Habit Analyzer in `src/lib/jarvis.ts`

Add this function:

```typescript
interface HabitPattern {
  domain: string
  completionRate: number
  streak: number
  lastActive: string | null
  trend: 'improving' | 'declining' | 'consistent' | 'inactive'
}

async function analyzeHabits(
  tasks?: Array<{ task_date: string, title: string, completed: boolean, description: string | null }>
): Promise<string> {
  try {
    // If tasks not provided, fetch last 7 days
    if (!tasks) {
      const weekAgo = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000)
        .toLocaleDateString('en-CA')

      const { data } = await supabase
        .from('horizon_tasks')
        .select('task_date, title, completed, description')
        .gte('task_date', weekAgo)
        .order('task_date', { ascending: true })

      tasks = data ?? []
    }

    if (tasks.length === 0) return 'No task history available for habit analysis.'

    // Timeline domain keywords for classification
    const DOMAIN_KEYWORDS: Record<string, string[]> = {
      gym: ['gym', 'workout', 'exercise', 'fitness', 'strength', 'cardio', 'training'],
      development: ['code', 'dev', 'build', 'implement', 'javascript', 'react', 'project'],
      cricket: ['cricket', 'batting', 'bowling', 'practice', 'match'],
      freelance: ['client', 'freelance', 'project', 'deliver', 'invoice'],
      study: ['study', 'learn', 'revision', 'exam', 'course'],
      ai: ['ai', 'jarvis', 'ml', 'model', 'gemini'],
    }

    // Classify tasks by domain
    const domainStats: Record<string, { total: number, completed: number, dates: string[] }> = {}

    for (const task of tasks) {
      const titleLower = task.title.toLowerCase()
      let domain = 'general'

      for (const [d, keywords] of Object.entries(DOMAIN_KEYWORDS)) {
        if (keywords.some(k => titleLower.includes(k))) {
          domain = d
          break
        }
      }

      // Also check Timeline prefix in description
      if (task.description) {
        const match = task.description.match(/\[timeline:[^:]+:([^\]]+)\]/)
        if (match) domain = match[1]
      }

      if (!domainStats[domain]) {
        domainStats[domain] = { total: 0, completed: 0, dates: [] }
      }

      domainStats[domain].total++
      if (task.completed) {
        domainStats[domain].completed++
        if (!domainStats[domain].dates.includes(task.task_date)) {
          domainStats[domain].dates.push(task.task_date)
        }
      }
    }

    // Build insight string
    const insights: string[] = []

    for (const [domain, stats] of Object.entries(domainStats)) {
      if (stats.total < 2) continue // not enough data

      const rate = Math.round((stats.completed / stats.total) * 100)
      const activeDays = stats.dates.length

      if (rate === 0 && stats.total >= 2) {
        insights.push(`${domain}: 0% completion this week (${stats.total} tasks skipped)`)
      } else if (rate >= 80) {
        insights.push(`${domain}: strong week — ${rate}% completion, active ${activeDays} days`)
      } else if (rate < 40 && stats.total >= 3) {
        insights.push(`${domain}: low engagement — only ${rate}% done this week`)
      }
    }

    return insights.length > 0
      ? `Weekly habit patterns:\n${insights.join('\n')}`
      : 'Consistent effort across all areas this week.'
  } catch {
    return ''
  }
}
```

```typescript
      )
      ttsQueue.enqueue('Saved.')
      return `Noted. ${taskResult}`

    return 'Noted.'
  } catch {
    ttsQueue.enqueue('Saved.')

  }
```

### Step 2 — Batch Task Function in `src/lib/horizon.ts`

Add after existing `addTaskDirect`:

```typescript
export async function addTasksBatch(
  tasks: Array<{
    title: string
    taskDate: string
    taskTime: string
    priority: 'low' | 'medium' | 'high'
    description?: string
  }>
): Promise<{ created: number, failed: number }> {
  let created = 0
  let failed = 0

  // Insert all tasks in parallel
  const results = await Promise.allSettled(
    tasks.map(task => addTaskDirect(task))
  )

  for (const result of results) {
    if (result.status === 'fulfilled') {
      created++
    } else {
      failed++
      console.error('[Horizon] Batch task failed:', result.reason)
    }
  }

  return { created, failed }
}
```

### Step 3 — Handle In `executeToolCall` in `src/lib/gemini.ts`

Add handler:

```typescript
case 'create_tasks_batch': {
  const { tasks } = toolArgs as {
    tasks: Array<{
      title: string
      taskDate: string
      taskTime: string
      priority: 'low' | 'medium' | 'high'
      description?: string
    }>
  }

  if (!tasks || tasks.length === 0) {
    return 'No tasks provided.'
  }

  const { created, failed } = await addTasksBatch(tasks)

  if (failed === 0) {
    return `Created ${created} task${created > 1 ? 's' : ''} successfully.`
  } else {
    return `Created ${created} tasks. ${failed} failed — check your Horizon page.`
  }
}
```
